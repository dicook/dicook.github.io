recent_grad <- read.csv(text=getURL("https://raw.githubusercontent.com/fivethirtyeight/data/master/college-majors/recent-grads.csv"))
grads <- merge(grad_data, recent_grad, by = "Major_code", all.x = TRUE)

names(grads) <- sub("ShareWomen", "x", names(grads))
names(grads) <- sub("Nongrad_median", "y", names(grads))
names(grads) <- sub("Grad_median", "y2", names(grads))

p <- qplot(x, y, data = grads) + geom_smooth(method = "lm")

p2 <- qplot(x, y2, data = grads) + geom_smooth(method = "lm")


# http://stackoverflow.com/questions/7549694/ggplot2-adding-regression-line-equation-and-r2-on-graph
lm_eqn = function(df){
  m = lm(y ~ x, df);
  eq <- substitute(italic(y) == a + b %.% italic(x)*","~~italic(r)^2~"="~r2, 
                   list(a = format(coef(m)[1], digits = 2), 
                        b = format(coef(m)[2], digits = 2), 
                        r2 = format(summary(m)$r.squared, digits = 3)))
  as.character(as.expression(eq));                 
}

lm_eqn2 = function(df){
  m = lm(y2 ~ x, df);
  eq <- substitute(italic(y) == a + b %.% italic(x)*","~~italic(r)^2~"="~r2, 
                   list(a = format(coef(m)[1], digits = 2), 
                        b = format(coef(m)[2], digits = 2), 
                        r2 = format(summary(m)$r.squared, digits = 3)))
  as.character(as.expression(eq));                 
}

p + geom_text(aes(x = 0.75, y = 120000, label = lm_eqn(grads)), parse = TRUE) +
  xlab("ShareWomen") + ylab("Nongrad_median")
  
p2 + geom_text(aes(x = 0.75, y = 120000, label = lm_eqn2(grads)), parse = TRUE) +
  xlab("ShareWomen") + ylab("Grad_median")
